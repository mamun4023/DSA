
<script type="text/javascript">
	
	class Node{
		constructor(element){
			this.element = element;
			this.next = null;
		}
	}


   class LinkedList{
   	 constructor(element){
   	 	this.head = element;
   	 	this.size = 0;
   	 }
   }


   // inserting an element

   function add(element){
   		var node = new Node(element);
   		var traverse;
   		if(this.head == null)
   			this.head = node;
   		else{
   			traverse = this.head;
   			while(traverse.next){
   				traverse = traverse.next;
   			}
   			traverse.next = node;
   		}
   		this.size++;
   }

add(10);
add(30);



// printing  list

function printList(){
	var traverse = this.head;
	while(traverse){
		document.write(traverse.element + " ");
		traverse = traverse.next;
	}
}

printList();


// remove an element 

function removeElement(element){
	var traverse = this.head;
	var prev = null;

	while(traverse != null){
		if(traverse.element === element){
			if(prev == null){
				this.head  = traverse.next;
			}else{
				prev.next = traverse.next;
			}
			this.size--;
			return traverse.element;
		}
		prev = traverse;
		traverse = traverse.next;
	}
	
	return -1;
}

document.write("<br> After removed </br>")
removeElement(10);
printList();

</script>
