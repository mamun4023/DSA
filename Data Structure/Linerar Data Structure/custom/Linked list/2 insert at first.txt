<script type="text/javascript">
	

	// create a node

	class Node{
		constructor(data){
			this.data = data;
			this.next = null;
		}
	}


	class Linked_list{
            constructor(){
            	this.head = null;  
            }
		

		// add data at first of linked list
		insert(data){
		
			var after_head = this.head;
			this.head = new Node(data);
			this.head.next = after_head;
		}

		// print Linked list

		print(){
			var traverse = this.head;
			while(traverse){
				document.write(traverse.data +" ");
				traverse= traverse.next;
			}
		}
	}

 obj = new Linked_list();
 obj.insert(10);
 obj.insert(30);
 obj.insert(100);
 obj.print();



</script>
